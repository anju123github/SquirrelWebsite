9. Program to find the sum of each row and each column of a matrix

Code


                
import java.util.Scanner;

public class SumOfRowsAndColumns {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
                                
        // Get the dimensions of the matrix
        System.out.print("Enter the number of rows: ");
        int rows = scanner.nextInt();
                                
        System.out.print("Enter the number of columns: ");
        int columns = scanner.nextInt();
                                
        // Create a matrix
        int[][] matrix = new int[rows][columns];
                                
        // Get elements for the matrix
        enterMatrixElements(scanner, matrix);
                                
        // Calculate and display the sum of each row and each column
        System.out.println("Original Matrix:");
        printMatrix(matrix);
                                
        calculateAndDisplaySumOfRows(matrix);
        calculateAndDisplaySumOfColumns(matrix);
                                
        scanner.close();
    }
                                
    // Function to enter matrix elements
    private static void enterMatrixElements(Scanner scanner, int[][] matrix) {
        System.out.println("Enter elements for the matrix:");
                                
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print("Enter element at position [" + i + "][" + j + "]: ");
                matrix[i][j] = scanner.nextInt();
            }
        }
    }
                                
    // Function to calculate and display the sum of each row
    private static void calculateAndDisplaySumOfRows(int[][] matrix) {
        System.out.println("Sum of Each Row:");
                                
        for (int i = 0; i < matrix.length; i++) {
            int rowSum = 0;
            for (int j = 0; j < matrix[0].length; j++) {
                rowSum += matrix[i][j];
            }
            System.out.println("Row " + (i + 1) + ": " + rowSum);
        }
    }
                                
    // Function to calculate and display the sum of each column
    private static void calculateAndDisplaySumOfColumns(int[][] matrix) {
        System.out.println("Sum of Each Column:");
                                
        for (int j = 0; j < matrix[0].length; j++) {
            int colSum = 0;
            for (int i = 0; i < matrix.length; i++) {
                colSum += matrix[i][j];
            }
            System.out.println("Column " + (j + 1) + ": " + colSum);
        }
    }
                                
    // Function to print a matrix
    private static void printMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println(); // Move to the next row
        }
        System.out.println(); // Separate matrices by a blank line
    }
}
