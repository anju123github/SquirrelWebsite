26. Write a Java program to create an abstract class GeometricShape with abstract methods area() and perimeter(). Create subclasses Triangle and Square that extend the GeometricShape class and implement the respective methods to calculate the area and perimeter of each shape

Code


                
// Abstract class GeometricShape
abstract class GeometricShape {
    // Abstract methods
    public abstract double area();
    public abstract double perimeter();
}
                                        
// Subclass Triangle extending GeometricShape
class Triangle extends GeometricShape {
    // Fields for the sides of the triangle
    private double sideA;
    private double sideB;
    private double sideC;
                                        
    // Constructor for Triangle
    public Triangle(double sideA, double sideB, double sideC) {
        this.sideA = sideA;
        this.sideB = sideB;
        this.sideC = sideC;
    }
                                        
    // Implementing the area method for Triangle
    
    public double area() {
        // Using Heron's formula to calculate the area of the triangle
        double s = (sideA + sideB + sideC) / 2;
        return Math.sqrt(s * (s - sideA) * (s - sideB) * (s - sideC));
    }
                                        
    // Implementing the perimeter method for Triangle
    
    public double perimeter() {
        return sideA + sideB + sideC;
    }
}
                                        
// Subclass Square extending GeometricShape
class Square extends GeometricShape {
    // Field for the side of the square
    private double side;
                                        
    // Constructor for Square
    public Square(double side) {
        this.side = side;
    }
                                        
    // Implementing the area method for Square
    
    public double area() {
        return side * side;
    }
                                        
    // Implementing the perimeter method for Square
    
    public double perimeter() {
        return 4 * side;
    }
}
                                        
// Main class
public class AbstractGeometricShapeExample {
    public static void main(String[] args) {
        // Create instances of Triangle and Square
        Triangle triangle = new Triangle(3, 4, 5);
        Square square = new Square(4);
                                        
        // Calculate and display the area and perimeter for each shape
        System.out.println("Triangle:");
        System.out.println("Area: " + triangle.area());
        System.out.println("Perimeter: " + triangle.perimeter());
                                        
        System.out.println(); // Empty line for clarity
                                        
        System.out.println("Square:");
        System.out.println("Area: " + square.area());
        System.out.println("Perimeter: " + square.perimeter());
    }
}
