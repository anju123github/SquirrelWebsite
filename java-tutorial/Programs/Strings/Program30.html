30. Program to find the number of the words in the given text file

Code


                
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class WordCount {
    public static void main(String[] args) {
        String filePath = "path/to/your/textfile.txt";
        // Calling the method to find the number of words
        int wordCount = countWords(filePath);

        // Displaying the result
        System.out.println("Number of Words: " + wordCount);
    }

   public static int countWords(String filePath) {
        int wordCount = 0;

        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            // Reading the file line by line
            String line;
            while ((line = br.readLine()) != null) {
                // Splitting each line into words
                String[] words = line.split("\\s+");

                // Counting the number of words
                wordCount += words.length;
            }
        } catch (IOException e) {
            e.printStackTrace();
        }

        return wordCount;
    }
}

Output

    Number of Words: 28
