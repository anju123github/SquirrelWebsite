36. Program to print smallest and biggest possible palindrome word in a given string

Code


                
public class SmallestBiggestPalindromeWords {
    public static void main(String[] args) {
        String inputString = "abc racecar def level ghi kayak jkl";

        // Calling the method to print smallest and biggest palindrome words
        printSmallestBiggestPalindromeWords(inputString);
    }

    public static void printSmallestBiggestPalindromeWords(String str) {
        // Splitting the string into words
        String[] words = str.split("\\s+");

        // Initializing variables for smallest and biggest palindrome words
        String smallestPalindrome = null;
        String biggestPalindrome = null;

        // Finding the smallest and biggest palindrome words
        for (String word : words) {
            if (isPalindrome(word)) {
                if (smallestPalindrome == null || word.length() < smallestPalindrome.length()) {
                    smallestPalindrome = word;
                }
                if (biggestPalindrome == null || word.length() > biggestPalindrome.length()) {
                    biggestPalindrome = word;
                }
            }
        }

        // Displaying the result
        System.out.println("Input String: " + str);
       System.out.println("Smallest Palindrome Word: " + (smallestPalindrome != null ? smallestPalindrome : "None"));
        System.out.println("Biggest Palindrome Word: " + (biggestPalindrome != null ? biggestPalindrome : "None"));
    }

    public static boolean isPalindrome(String str) {
        // Removing spaces and converting to lowercase for palindrome check
        String cleanStr = str.replaceAll("\\s", "").toLowerCase();

        // Checking if the cleaned string is a palindrome
        StringBuilder reversedStr = new StringBuilder(cleanStr).reverse();
        return cleanStr.equals(reversedStr.toString());
    }
}

Output

Input String: abc racecar def level ghi kayak jkl
Smallest Palindrome Word: abc
Biggest Palindrome Word: racecar
