22. Program to find maximum and minimum occurring character in a string

Code


            
import java.util.HashMap;
import java.util.Map;

public class MaxMinOccurringCharacter {
    public static void main(String[] args) {
        String inputString = "hello world";

        // Calling the method to find max and min occurring characters
        findMaxMinCharacters(inputString);
    }

    public static void findMaxMinCharacters(String str) {
        // Removing spaces and converting the string to lowercase for case-insensitive comparison
        str = str.replaceAll("\\s", "").toLowerCase();

        // Creating a map to store character frequencies
        Map charFrequencyMap = new HashMap<>();

        // Populating the map with character frequencies
        for (char c : str.toCharArray()) {
            charFrequencyMap.put(c, charFrequencyMap.getOrDefault(c, 0) + 1);
        }

        // Finding the maximum occurring character
        char maxChar = '\0'; // Placeholder for maxChar
        int maxFrequency = Integer.MIN_VALUE;

        // Finding the minimum occurring character
        char minChar = '\0'; // Placeholder for minChar
        int minFrequency = Integer.MAX_VALUE;

        for (Map.Entry entry : charFrequencyMap.entrySet()) {
            char currentChar = entry.getKey();
            int currentFrequency = entry.getValue();

            // Updating maxChar and maxFrequency
            if (currentFrequency > maxFrequency) {
                maxChar = currentChar;
                maxFrequency = currentFrequency;
            }

            // Updating minChar and minFrequency
            if (currentFrequency < minFrequency) {
                minChar = currentChar;
                minFrequency = currentFrequency;
            }
        }

        // Displaying the result
        System.out.println("Input String: " + str);
        System.out.println("Maximum Occurring Character: " + maxChar + " (Frequency: " + maxFrequency + ")");
        System.out.println("Minimum Occurring Character: " + minChar + " (Frequency: " + minFrequency + ")");
    }
}
