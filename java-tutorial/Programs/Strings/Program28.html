28. Program to find the longest repeating sequence in a string

Code


                
public class LongestRepeatingSequence {
    public static void main(String[] args) {
        String inputString = "ABABC";

        // Calling the method to find the longest repeating sequence
        String longestRepeatingSequence = findLongestRepeatingSequence(inputString);

        // Displaying the result
        System.out.println("Input String: " + inputString);
        System.out.println("Longest Repeating Sequence: " + longestRepeatingSequence);
    }

    public static String findLongestRepeatingSequence(String str) {
       int n = str.length();
        String result = "";

        // Building a 2D array to store the lengths of the longest common suffixes
        int[][] dp = new int[n + 1][n + 1];

        // Populating the dp array
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
               if (str.charAt(i - 1) == str.charAt(j - 1) && dp[i - 1][j - 1] < (j - i)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;

                    if (dp[i][j] > dp[i][n]) {
                        int len = dp[i][j];
                        result = str.substring(i - len, i);
                    }
                } else {
                    dp[i][j] = 0;
                }
            }
        }

        return result;
   }
}
