1. Enter the character from user using InputStreamReader

Code


                
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class ReadCharFromUser {
    public static void main(String[] args) {
        // Create an InputStreamReader to read characters from the console
        InputStreamReader inputStreamReader = new InputStreamReader(System.in);

        // Wrap the InputStreamReader with BufferedReader for easier reading
        BufferedReader bufferedReader = new BufferedReader(inputStreamReader);

        try {
            // Prompt the user to enter a character
            System.out.print("Enter a character: ");

            // Read a single character from the user
            char userInput = (char) bufferedReader.read();

            // Print the entered character
            System.out.println("You entered: " + userInput);
        } catch (IOException e) {
            System.err.println("Error reading input: " + e.getMessage());
        } finally {
            try {
                // Close the BufferedReader
                bufferedReader.close();
            } catch (IOException e) {
                System.err.println("Error closing BufferedReader: " + e.getMessage());
            }
        }
    }
}

Output

Enter a character: A

You entered: A
